## datetime

```python
import datetime

print('\n--------------現在時間--------------')

now = datetime.datetime.now()
# 通常表記
print(now)
# 国際規格表記（ISO Format）
print(now.isoformat())
# 任意のフォーマット
print(now.strftime('%Y/%m/%d-%H:%M:%S%f'))

print('\n--------------今日--------------')

today = datetime.datetime.today()
# 通常表記
print(today)
# 国際規格表記（ISO Format）
print(today.isoformat())
# 任意のフォーマット
print(today.strftime('%Y/%m/%d'))

print('\n--------------任意の時間--------------')

# 時間を生成
t = datetime.time(hour=1, minute=10, second=5, microsecond=100)
print(t)
print(t.isoformat())
print(now.strftime('%H:%M:%S%f'))

print('\n--------------1週間前--------------')
# 1週間を引く
d = datetime.timedelta(weeks=1)
print(now - d)

# 1週間`前`を足す
d = datetime.timedelta(weeks=-1)
print(now + d)
```

### timeをつかったファイルのバックアップ
```python
import os
import shutil
import time

file_name = 'test.txt'

# ファイルが存在したら、時刻をつけたファイル名にして、ファイルコピー
if os.path.exists(file_name):
    shutil.copy(file_name,
                f'{now.strftime("%Y%m%d_%H%M%S")}_{file_name}')

# ファイルが無ければ生成
with open(file_name, 'w') as f:
    f.write('test')
```
